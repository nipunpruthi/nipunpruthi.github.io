sudo systemctl enable docker

```

```

```
docker images
docker build . // after writing the dockerfile
docker build . -t nipun_node:latest // custom naming
docker run -it image_id /bin/sh

ctrl+pq, deattach from container, it will continue to work in background
docker stop container id

docker run -p 8000:8000 -it containerID /bin/sh
docker run -d(deattach) -p 8000:8000 containerID
```
